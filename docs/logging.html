# Logging

Logging can be configured in a fashion similar to environment variables: global, local, remote, per server and per component.

# Top level

export default {
  logs: {
    global: 'logs/global.log',
    local: 'logs/local.log',
    ssh: 'logs/ssh.log'
  },
}

# Per server

export default {
  ssh: {
    server1: {
      hostname: '1.2.3.4',
      username: 'serveruser',
      log: 'logs/ssh-server1.log'
    }
  }
}

# Per component

<fabula>
export default {
  log: 'logs/component.log'
}
</fabula>
